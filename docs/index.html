Quickstart

Run in Five Minutes

Install 
# Clone and install
git clone https://github.com/Organica-Ai-Solutions/EntropicUnification.git
cd EntropicUnification
python -m venv venv && source venv/bin/activate
pip install -r requirements.txt
Schwarzschild Test 
# Run the flagship experiment
python examples/schwarzschild_test.py \
  --iterations 1000 --lattice 64 \
  --formulation massless --device auto

# Entanglement scaling sweep
python examples/scaling_experiment.py \
  --iterations 1000 --device auto
Python API 
from core.quantum_engine import QuantumEngine
from core.geometry_engine import GeometryEngine
from core.entropy_module import EntropyModule
from core.coupling_layer import CouplingLayer
from core.optimizer import EntropicOptimizer

# Initialize
qe = QuantumEngine(QuantumConfig(num_qubits=4, depth=2))
ge = GeometryEngine(dimensions=2, lattice_size=32)
em = EntropyModule(qe)
cl = CouplingLayer(ge, em,
    stress_form=StressTensorFormulation.MASSLESS)

# Run Schwarzschild test
from examples.schwarzschild_test import run_schwarzschild_test
results = run_schwarzschild_test({
    "lattice_size": 32, "n_iterations": 300,
    "formulation": "massless", "device": "auto"
})
Dashboard 
# Interactive dashboard
python dashboards/run_fixed_dashboard.py

# Compare formulations
python examples/compare_stress_tensors.py
